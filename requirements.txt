# Colab-only requirements — do not `pip install -r requirements.txt` locally
# (this repo does not run on the laptop; see CLAUDE.md).
#
# These versions are pinned for Colab REPRODUCIBILITY: a fresh Colab runtime can
# reinstall CUDA/drivers or update its base image between sessions, and an
# unpinned install would silently drift to whatever is newest that day.
#
# Pinning versions alone is NOT enough — install ORDER matters too:
#   torch -> vllm==0.8.3 -> flash-attn -> verl -> ray
# vLLM in particular will silently downgrade/replace the installed torch build
# if it is installed before torch, or without asserting torch survived.
# Do not `pip install -r requirements.txt` in one shot for this reason —
# always install via scripts/colab_install.sh, which enforces the order and
# checks torch's version after each step.
#
# Sources (verify against these before bumping any pin — verl moves fast):
#   https://github.com/volcengine/verl/blob/main/docs/README_vllm0.8.md
#   https://github.com/volcengine/verl/blob/main/requirements.txt
#   https://pypi.org/project/verl/

torch==2.6.0
vllm==0.8.3
flash-attn  # installed with --no-build-isolation by scripts/colab_install.sh
verl==0.8.0
ray[default]
